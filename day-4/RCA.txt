module RCA (input [0:3]a,b,
           input c,
           output [0:3]sum,
           output Carry);
wire c1,c2,c3;

FA M1(a[0],b[0],c,s[0],c1);
FA M2(a[1],b[1],c1,s[1],c2);
FA M3(a[2],b[2],c2,s[2],c3);
FA M4(a[3],b[3],c3,s[3],Carry);

endmodule
module FA (input a,b,c,
output sum,carry);

assign sum=a^b^c;
assign carry=(a^b)&c|(a&b);

endmodule
